题意翻译过来其实就是

以边的编号作为边权 给定 $q$ 次询问 $l,r$ 问让 $[l,r]$ 范围内的所有点全部连通的 这些边的最大边权最小是多少

所以我们很容易联想到 Kruskal 重构树

但问题在于 普通的 Kruskal 重构树只能解决两个点之间的瓶颈问题 现在要求一段区间内的所有点连通的瓶颈应该如何做

我们可以画图发现规律

以下图为例 其中 $1 - 7 $ 号节点为原图中的节点 $8 - 13$ 号节点为新图中的节点 蓝色部分代表重构树上的点的点权

假设我们现在询问的是 $1$ 和 $5$ 两个节点 也就是要让 $1,2,3,4,5$ 全部连通 显然我们的代价是 $7$ 因为要让 $2$ 和 $3$ 连通必须有 $7$

可以发现 我们**只选择两个点** 就能代表所有节点 这两个点满足他们的 $LCA$ 一定是深度最低的

那么我们要选择什么样的点? 根据重构树的性质 我们要选择的一定是**叶子结点中最靠左和最靠右的点**

换句话说 如果我们给重构树上的节点标上 $dfs$ 序 选择的是 **$dfn$ 序最小和最大**的两个节点

![](https://raw.githubusercontent.com/unicornshjl/MyPic/img/test/b19eb429488221aa6d5e1eb8f2bc54b6.jpg)

所以虽然有 $[l,r]$ 范围内的 $r - l + 1$ 个点 但是实际上我们只需要利用 $dfn$ 序最小和最大的两个点 就可以等效这所有点

所以我们直接求这两个点的 $LCA$ 的点权即可

```cpp
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
#include<cmath>
#include<cstring>
#include<string>
#include<set>
#include<unordered_set>
#include<unordered_map>
#include<queue>
#include<cstdlib>
#include<iomanip>
#include<climits>
#include<bitset>

using namespace std;

typedef long long ll;

constexpr int maxn = 2e5 + 10;
constexpr int N = 21;
constexpr int mod = 1e9 + 7;
constexpr ll inf = 2e14;
constexpr ld eps = 1e-8;

int n, m, Q, f[maxn], cnt, val[maxn], tim, dfn[maxn], fa[maxn][N], dep[maxn], to[maxn];
int st_min[maxn][N], st_max[maxn][N];


struct E {
    int u, v, w;
} e[maxn];

vector<int> edge[maxn];

void add(int u, int v) {
    edge[u].push_back(v);
}

int find(int x) {
    return(x == f[x] ? x : f[x] = find(f[x]));
}

void dfs(int u, int father) {
    dfn[u] = ++tim; to[tim] = u;
    dep[u] = dep[father] + 1;
    fa[u][0] = father;
    for (int i = 1; i < N; i++)
        fa[u][i] = fa[fa[u][i - 1]][i - 1];
    for (auto v : edge[u]) {
        if (v == father) continue;
        dfs(v, u);
    }
}

void ST() {
    for (int i = 1; i <= n; i++)
        st_min[i][0] = st_max[i][0] = dfn[i];
    for (int i = 1; i < N; i++) 
        for (int j = 1; j + (1 << i) - 1 <= n; j++) {
            st_min[j][i] = min(st_min[j][i - 1], st_min[j + (1 << (i - 1))][i - 1]);
            st_max[j][i] = max(st_max[j][i - 1], st_max[j + (1 << (i - 1))][i - 1]);
        }
}

int querymin(int l, int r) {
    int len = log2(r - l + 1);
    return min(st_min[l][len], st_min[r - (1 << len) + 1][len]);
}

int querymax(int l, int r) {
    int len = log2(r - l + 1);
    return max(st_max[l][len], st_max[r - (1 << len) + 1][len]);
}

int get_lca(int u,int v) {
    if (dep[u] < dep[v]) swap(u, v);
    int k = dep[u] - dep[v];
    for (int i = N - 1; i >= 0; i--)
        if ((k >> i) & 1) u = fa[u][i];
    if (u == v) return u;
    for (int i = N - 1; i >= 0; i--)
        if (fa[u][i] != fa[v][i])
            u = fa[u][i], v = fa[v][i];
    return fa[u][0];
}

void solve(){
    cin >> n >> m >> Q;
    for (int i = 1; i <= m; i++) {
        cin >> e[i].u >> e[i].v;
        e[i].w = i;
    }
    for (int i = 1; i <= n; i++) f[i] = i;
    cnt = n;
    for (int i = 1; i <= m; i++) {
        int u = find(e[i].u), v = find(e[i].v);
        if (u == v) continue;
        cnt++;
        f[u] = f[v] = f[cnt] = cnt;
        add(cnt, u); add(cnt, v);
        val[cnt] = i;
    }
    tim = 0;
    for (int i = 1; i <= cnt; i++)
        if (f[i] == i)
            dfs(i, 0);
    ST();
    while (Q--) {
        int u, v;
        cin >> u >> v;
        int l = querymin(u, v), r = querymax(u, v);
        int lca = get_lca(to[l], to[r]);
        cout << val[lca] << " ";
    }
    for (int i = 1; i <= cnt; i++)
        edge[i].clear(), val[i] = 0;
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
