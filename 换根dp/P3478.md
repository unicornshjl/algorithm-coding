本题中的换根逻辑如上所示：当根从 $4$ 转移到 $5$ 时，绿色部分到根的距离全部 $-1$ 而粉色部分到根的距离全部 $+1。$

也就表明，当根从 $u$ 转移到其子节点 $v$ 时候，$v$ 的子树内部节点到根距离全部 $-1$ 而其余所有节点到根距离全部 $+1。$

那么我们已知 $dp_u$ 的情况下 $dp_v = dp_u - siz_v + n - siz_v。$

```cpp
#include<algorithm>
#include<bitset>
#include<climits>
#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iomanip>
#include<iostream>
#include<map>
#include<queue>
#include<set>
#include<string>
#include<unordered_map>
#include<unordered_set>
#include<vector>
#include<random>

using namespace std;

typedef long long ll;
constexpr ll inf = 1e18;
constexpr ld eps = 1e-8;
constexpr int intinf = 1e9 + 1e8;

constexpr int maxn = 1e6 + 10;
constexpr int N = 20;
constexpr int Base = 13;
constexpr int dx[] = { 0,0,1,-1 };
constexpr int dy[] = { 1,-1,0,0 };

int n;
ll siz[maxn], sum[maxn], dp[maxn];

vector<int> edge[maxn];

void add(int u, int v) {
    edge[u].push_back(v);
}

void dfs1(int u, int fa) {
    siz[u] = 1;
    for (auto v : edge[u]) {
        if (v == fa) continue;
        dfs1(v, u);
        siz[u] += siz[v];
        sum[u] += sum[v] + siz[v];
    }
    //以上为第一次 dfs , 由换根逻辑可知我们需要知道 u 的子树大小 siz[u]
    //而我们为了得到 u 的答案 , 所以也需要用 sum[u] 来保存 u 子树内所有节点到 u 的距离
}

void dfs2(int u, int fa) {
    for (auto v : edge[u]) {
        if (v == fa) continue;
        dp[v] = dp[u] - siz[v] + n - siz[v];
        dfs2(v, u);
    }
    //以上为第二次 dfs , 执行换根逻辑
}

void solve() {
    cin >> n;
    for (int i = 1; i < n; i++) {
        int u, v;
        cin >> u >> v;
        add(u, v); add(v, u);
    }
    dfs1(1, 0);//以 1 为根进行第一次 dfs
    dp[1] = sum[1];
    dfs2(1, 0);//以 1 为根进行第二次 dfs
    ll MAX = 0;//以下为找答案
    int ans;
    for (int i = 1; i <= n; i++) {
        if (dp[i] > MAX) {
            MAX = dp[i]; ans = i;
        }
    }
    cout << ans << "\n";
}

signed main() { 
    ios::sync_with_stdio(false); cin.tie(nullptr);
    solve();
    return 0;
}
```
